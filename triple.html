Projet 3: Application de convertion de fichiers géospatiales

Exemple du type d'interface possible

Présentation du code source

import customtkinter as ctk
				from tkinter import filedialog
				import subprocess

				# Initialize the main window
				app = ctk.CTk()
				app.title("Geospatial File Converter")
				app.geometry("400x200")

				# Function to browse for input file
				def browse_input_file():
					input_file.set(filedialog.askopenfilename())

				# Function to browse for output file
				def browse_output_file():
					output_file.set(filedialog.asksaveasfilename())

				# Function to run the ogr2ogr conversion
				def run_conversion():
					input_path = input_file.get()
					output_path = output_file.get()
					output_format = output_format_combobox.get()

					# Construct the ogr2ogr command
					ogr2ogr_command = [
						"ogr2ogr",
						"-f", output_format,
						output_path,
						input_path
					]

					# Execute the command
					subprocess.run(ogr2ogr_command)

				# Variables for file paths
				input_file = ctk.StringVar()
				output_file = ctk.StringVar()

				# Input file selection
				ctk.CTkLabel(app, text="Input File:").pack(pady=5)
				ctk.CTkEntry(app, textvariable=input_file, width=300).pack(pady=5)
				ctk.CTkButton(app, text="Browse", command=browse_input_file).pack(pady=5)

				# Output file selection
				ctk.CTkLabel(app, text="Output File:").pack(pady=5)
				ctk.CTkEntry(app, textvariable=output_file, width=300).pack(pady=5)
				ctk.CTkButton(app, text="Browse", command=browse_output_file).pack(pady=5)

				# Output format selection
				ctk.CTkLabel(app, text="Output Format:").pack(pady=5)
				output_format_combobox = ctk.CTkComboBox(app, values=["ESRI Shapefile", "GeoJSON", "GPKG"])
				output_format_combobox.pack(pady=5)

				# Run conversion button
				ctk.CTkButton(app, text="Convert", command=run_conversion).pack(pady=20)

				# Run the main loop
				app.mainloop()

Caractéristiques de l'application :

Interface graphique moderne utilisant CustomTkinter pour un rendu visuel amélioré

Sélection simple des fichiers via les boîtes de dialogue natives

Prise en charge de multiples formats de données géospatiales

Intégration directe avec GDAL/OGR pour des conversions fiables
